32 ((stat) >= 0 && (stat) <= (signed)lastEpicsAlarmCond) ? \
33 epicsAlarmConditionStrings[stat] : "??"

35#define sevr_to_str(stat) \
36 ((stat) >= 0 && (stat) <= (signed)lastEpicsAlarmSev) ? \

44 ((stat) <= lastEpicsAlarmSev) ? \
