223 bool m_prepare_switch(int engine, CumbiaPool *cu_pool, CuControlsFactoryPool &m_ctrl_factory_pool, std::vector<Cumbia *>& cumbias) {
224#if !defined(CUMBIA_HTTP_VERSION) && !defined(QUMBIA_TANGO_CONTROLS_VERSION)
225 d->error = true;
226 d->msg = "neither Tango nor Http modules are compiled into the library";
227#else
228 d->error = false;
229#endif
230 // supported engines: Tango and Http
231 d->error = (engine != CumbiaHttp::CumbiaHTTPType && engine != CumbiaTango::CumbiaTangoType);
232 if(d->error)
233 d->msg = "engine hot switch supports the Tango and Http engines only";
234 else {
235 d->error = same_engine(cu_pool, engine);
236 if(d->error) {
237 d->msg = "trying to switch to the same engine";
238 }
239 else { // switch to the new engine
240 // save the log instance
241 // when engine switching, we suppose there is only one Cumbia in the pool
242 // because interchangeable engines are mutually exclusive (see cclear.size() == 1 below)
243 //
244 CuLog *log = nullptr;
245 Cumbia *cu = nullptr; // new cumbia
246 if(engine == CumbiaTango::CumbiaTangoType) {
247 CuTangoRegisterEngine tare;
248 cumbias = m_clear_cumbia(cu_pool, m_ctrl_factory_pool);
249 if(cumbias.size() == 1 && cumbias[0]->getServiceProvider()) // recycle log
250 log = static_cast<CuLog *>(cumbias[0]->getServiceProvider()->get(CuServices::Log));
251 cu = tare.registerWithDefaults(cu_pool, m_ctrl_factory_pool);
252 }
253 else if(engine == CumbiaHttp::CumbiaHTTPType) {
254 CuHttpRegisterEngine httpre;
255 d->error = !httpre.load(qApp->arguments(), true); // true: try loading also without -u in args
256 if(!d->error) {
257 cumbias = m_clear_cumbia(cu_pool, m_ctrl_factory_pool);
258 if(cumbias.size() == 1 && cumbias[0]->getServiceProvider()) // recycle log
259 log = static_cast<CuLog *>(cumbias[0]->getServiceProvider()->get(CuServices::Log));
260 cu = httpre.registerWithDefaults(cu_pool, m_ctrl_factory_pool);
261 }
262 else
263 d->msg = "error registering http module: valid URL '" + httpre.url() + "' ?";
264 }
265 for(const std::string &n : cu_pool->names()) {
266 printf("+ cumbia registered %s -> %p type %d\n", n.c_str(), cu_pool->get(n), cu_pool->get(n)->getType());
267 }
268
269 // install log instance saved from earlier cumbia
270 if(cu && log)
271 cu->getServiceProvider()->registerSharedService(CuServices::Log, log);
272 else
273 pretty_pri("\033[1;33m*\033[0m: no log service installed in new cumbia: no log service found in older instance");
274 } // !d->error trying to switch to the same engine
275 } // ! error unsupported target engine
276 return !d->error;
277 }

304 int engine_type(const CuContext *ctx) const {
305 // get current engine in use
306 Cumbia *c = nullptr;
307 d->error = (ctx == nullptr);
308 if(!d->error) {
309 if(ctx->getReader())
310 c = ctx->getReader()->getCumbia();
311 else if(ctx->getWriter())

313 else {
314 d->error = true;

316 }
317 }
318 if(!d->error && !c) {
319 d->error = true;

336 d->error = false;

338 // clear the pool
339 for(size_t i = 0; i < cu_pool->names().size() && !d->error; i++) {
340 const std::string n = cu_pool->names().at(i);
341 d->error = !cu_pool->get(n);
342 if(!d->error) {
343 printf("unregistering cumbia impl '%s'\n", n.c_str());
344 vc.push_back(cu_pool->get(n));
345 cu_pool->unregisterCumbiaImpl(n);

347 }
348 else
349 d->msg = QString("error getting cumbia instance for \"%1\"").arg(n.c_str());
350 }
351 if(!d->error)
352 fp = CuControlsFactoryPool();
